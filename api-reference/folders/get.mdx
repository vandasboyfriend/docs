<RequestExample >

```js JavaScript
var myHeaders = new Headers();
myHeaders.append("X-Token", "YOUR_TOKEN_HERE");

var requestOptions = {
  method: 'GET',
  headers: myHeaders,
  redirect: 'follow'
};

fetch("https://api.empr.cloud/api/v1/folders", requestOptions)
  .then(response => response.text())
  .then(result => console.log(result))
  .catch(error => console.log('error', error));
```
```python Python
import requests

url = "https://api.empr.cloud/api/v1/folders"

payload = {}
headers = {
  'X-Token': 'YOUR_TOKEN_HERE'
}

response = requests.request("GET", url, headers=headers, data=payload)

print(response.text)
```
